  const distances = ['All', 'Under 1km', '1-5km', '5-10km', '10km+'];

      category: 'Educational',

      rating: 4.7,
      description: 'Interactive learning games for kids 3-7'

      image: 'https://placehold.co/200x200/3B82F6/white?text=Soccer',

        )}

                    alt={toy.name}
                    className="w-full h-full object-cover"
                  />
